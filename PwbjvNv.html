CSS Base

About CSS Base

It's a common practice to apply CSS to a page that styles elements such that they are consistent across all browsers. We offer two of the most popular choices: normalize.css and a reset. Or, choose Neither and nothing will be applied.

Vendor Prefixing

About Vendor Prefixing

To get the best cross-browser support, it is a common practice to apply vendor prefixes to CSS properties and values that require them to work. For instance -webkit- or -moz-.

We offer two popular choices: Autoprefixer (which processes your CSS server-side) and -prefix-free (which applies prefixes via a script, client-side).

        // 全局用户状态变量
        let currentUser = {
            isLoggedIn: false,
            username: "点击登录",
            avatar: "https://api.dicebear.com/7.x/bottts/svg?seed=Geography",
            isGuest: false

        // 1. 问答卡片展开与折叠
        function toggleAnswer(button) {
            const answer = button.nextElementSibling;
            if (answer.style.display === "block") {
                answer.style.display = "none";
                button.innerText = "显示答案";
            } else {
                answer.style.display = "block";
                button.innerText = "隐藏答案";
            }
        }

        // 2. 控制分册长条卡片丝滑展开
        function toggleVolume(headerElement) {
            const cardItem = headerElement.parentElement;
            cardItem.classList.toggle('active');
        }

        // 3. 身份认证面板控制
        function openUserModal() {
            document.getElementById("userModal").style.display = "flex";
            if(currentUser.isLoggedIn) {
                document.getElementById("modalTitle").innerText = "个人账户管理";
                document.getElementById("loginForm").style.display = "none";
                document.getElementById("profileForm").style.display = "block";
                document.getElementById("editName").value = currentUser.username;
            } else {
                document.getElementById("modalTitle").innerText = "极简身份认证";
                document.getElementById("loginForm").style.display = "block";
                document.getElementById("profileForm").style.display = "none";
            }
        }

        function closeUserModal(e) {
            document.getElementById("userModal").style.display = "none";
        }

        // 4. 执行登录操作
        function doLogin(isGuestMode) {
            if(isGuestMode) {
                currentUser.username = "地理游客";
                currentUser.avatar = "https://api.dicebear.com/7.x/bottts/svg?seed=Guest";
                currentUser.isGuest = true;
            } else {
                let account = document.getElementById("loginAccount").value.trim();
                if(!account) {
                    alert("请输入手机号或邮箱地址！");
                    return;
                }
                currentUser.username = account.substring(0, 5) + "...";
                currentUser.avatar = "https://api.dicebear.com/7.x/bottts/svg?seed=" + account;
                currentUser.isGuest = false;
            }
            currentUser.isLoggedIn = true;
            updateNavUserUI();
            closeUserModal();
        }

        // 5. 注销登录
        function doLogout() {
            currentUser.isLoggedIn = false;
            currentUser.username = "点击登录";
            currentUser.avatar = "https://api.dicebear.com/7.x/bottts/svg?seed=Geography";
            updateNavUserUI();
            closeUserModal();
        }

        // 6. 头像选择切换
        function selectAvatar(element) {
            document.querySelectorAll('.avatar-opt').forEach(el => el.classList.remove('selected'));
            element.classList.add('selected');
            tempSelectedAvatarUrl = element.src;
        }

        // 7. 保存并更新个人资料
        function saveProfile() {
            let newName = document.getElementById("editName").value.trim();
            if(newName) currentUser.username = newName;
            if(tempSelectedAvatarUrl) currentUser.avatar = tempSelectedAvatarUrl;
            
            updateNavUserUI();
            closeUserModal();
        }

        function updateNavUserUI() {
            document.getElementById("navName").innerText = currentUser.username;
            document.getElementById("navAvatar").src = currentUser.avatar;
        }

        // 8. 交流区发帖
        function publishPost() {
            let content = document.getElementById("forumInput").value.trim();

                alert("请输入你想要交流的内容。");
                return;
            }
            
            let authorName = currentUser.isLoggedIn ? currentUser.username : "匿名同学";
            let authorAvatar = currentUser.avatar;

                <div class="post-content">
                    <h4>${authorName}</h4>
                    <p>${content}</p>
                </div>
            `;
            
            forumList.insertBefore(newPost, forumList.firstChild);
            document.getElementById("forumInput").value = "";
        }
